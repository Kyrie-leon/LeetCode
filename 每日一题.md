## 1. 求最近的公共祖先

思路：

不难发现一个规律，a和b的祖先都是a或b/(除以)2，要么a和b有公共祖先，要么a是b的祖先或者b是a的祖先，这样很容易求解了。循环让最大的数字先除以2，因为有可能小的数字是大的数字的祖先，直到a和b相等就可以了

```c++
class LCA {
public:
    int getLCA(int a, int b) {
        // write code here
        while(a != b)
        {
            if(a > b)
                a /= 2;
            if(a < b)
                b /=2;
        }
        return b;
    }
};
```



## 2.求byte最大的连续数字1

思路：

1. 构建一个数组存放byte转化为2进值数字的每一位数字，然后定义一个计数器count，遍历这个数组并计数连续的1的个数

2. 直接使用位运算，让数字byte和1进行逻辑与运算，然后右移知道数字byte为0

   注意细节：牛客网的测试用例为循环输入，因此输入用例需要写一个while循环，切记切记切记！

```c++
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;

int MaxConNum(int n)
{
    int count = 0;
    int ret = 0;
    vector<int> nums;
    while(n)
    {
        nums.push_back(n%2);
        n /= 2;
    }
    
    for(auto num : nums)
    {
        if(num == 1)
        {
            count++;
            if(count > ret)
                ret = count;
        }
        else
        {
            count = 0;
        }
    }
    return ret;
}

int MaxConNum1(int n)
{
    int count = 0;
    int ret = 0;
    while(n)
    {
        if(n & 1)
        {
            ++count;
            if(count > ret)
                ret = count;
        }
        else
            count = 0;
        n >>= 1;
    }
    return ret;
}

int main()
{
    int n;
    while(cin >> n)
    {
        int ret = MaxConNum1(n);
        cout << ret << endl;
    }
```

## 最小素数对

思路：

从中间开始不断缩小，只要找到两个素数的和为n就可以打印

```c++
#include<iostream>
#include<math.h>
using namespace std;

bool IsPrime(int n)
{
    int i = 2;
    while(i <= sqrt(n))
    {
        if(n % i == 0)
            return false;
        i++;
    }
    
    return true;
}

void MaxPrimeNum(int n)
{
    int mid = n/2;
    int i = mid;
    for(; i >0; --i)
    {
        if(IsPrime(i) && IsPrime(n-i))
            break;
    }
    cout << i << endl;
    cout << n-i <<endl;
}

int main()
{
    int num;
    while(cin >> num)
    {
        MaxPrimeNum(num);
    }
    return 0;
}
```

```c++
void SegArg(string& s)
{
    int count = 0; //record paremeters
    int flag = 1; //mark ""
     
    //第一遍记录参数值
    for(int i = 0; i < s.size(); ++i)
    {
        //遇见空格就参数+1
        if(s[i] == ' ')   
            count++;
        //遇见双引号，先++i继续遍历直到碰到下一个“
        if(s[i] == '"')
        {
            i++;
            while(s[i] != '"')
            {
                i++;
            }
        }
    }
    cout << count+1 << endl; //空格少一个因此count+1
    //第二次遍历输出各个参数
    for(int i = 0; i < s.size(); ++i)
    {
        //遇见引号flag异或1
        if(s[i] == '"')
        {
            flag ^= 1;
        }
        //不是空格并且不为“就打印
        if((s[i] != ' ') && (s[i] != '"'))
            cout << s[i];
        //空格且引号内也打印
        if(s[i] == ' ' && (!flag))
            cout << s[i];
        //遇见空格且不在分号内直接换行
        if(s[i] == ' ' && flag)
            cout<< endl;
    }
}

{
    string arg;

    {
        SegArg(arg);
    }
```

## 小易跳石板

```c++
#include<iostream>
#include<vector>
#include<math.h>
using namespace std;

void divsorNum(int n, vector<int> &s)
{
    for(int i = 2; i <= sqrt(n); ++i)
    {
        if(n % i ==0)
        {
            s.push_back(i);
            if(n/i != i)
                s.push_back(n/i);
        }
    }
}

int MinJump(int n, int m)
{
    vector<int> StepNum(m+1,0);    //存储每个石板的步数
    StepNum[n] = 1;    //起始位置记为1
    //第一层遍历每个编号的石板记录步数
    //第二层遍历每个石板的编号的所有约数
    for(int i = n; i <=m; ++i)
    {
        vector<int> divNum;    //存储i的所有公约数
        if(StepNum[i] == 0)    //如果为0表示这个石板不可能到达，直接跳过
            continue;
        //求i的最大公约数
        divsorNum(i,divNum);
        
        for(int j = 0; j < divNum.size(); ++j)
        {
            if(divNum[j] + i <= m && StepNum[divNum[j]+i] != 0)
            {
                StepNum[divNum[j]+i] = min(StepNum[divNum[j]+i],StepNum[i]+1);
            }
            else if(divNum[j] + i <= m)
            {
                StepNum[divNum[j]+i] = StepNum[i]+1;
            }
        }
    }
    
    if(StepNum[m] == 0)
        return -1;
    return StepNum[m] - 1;//多算一次所以减一
}

int main()
{
    int n,m;
    while(cin >> n >> m)
    {
        int ret = MinJump(n, m);
        cout << ret << endl;
    }
    return 0;
}
```

## 字符串通配符

解题思路

1. 遇到？且相等就往后走
2. 遇到*就匹配
3. 最后检查，如果都遍历完则匹配上
4. 只能AC80%

```
#include<iostream>
#include<string>
using namespace std;

bool MatchRet(string &src, string &dst)
{
    //求出最大长度作为结束条件
    int srcLen = src.size();
    int dstLen = dst.size();
    int maxLen = srcLen > dstLen ? srcLen:dstLen;
    int i = 0;
    int j = 0;
    while(i < srcLen || j < dstLen)
    {
        if(src[i] == '?')
        {
            i++;
            j++;
        }
        else if(src[i] == '*')
        {
            i++;
            if( i != srcLen)
            {
                while(j < dstLen && src[i] != dst[j])
                {
                    ++j;
                }
            }
            else
                return true;

        }
        else if(src[i] != dst[j])
            return false;
        else
        {
            ++i;
            ++j;
        }
    }
    if(i != srcLen || j != dstLen)
        return false;
    return true;
}

int main()
{
    string src;
    string dst;
    while(cin >> src >> dst)
    {
        bool ret = MatchRet(src,dst);
        if(ret)
            cout << "true" <<endl;
        else
            cout << "false" << endl;
    }
    return 0;
}
```

## 洗牌

解题思路

可以发现规律123456洗完后142536，下标顺序为031425，因此只要再创建一个临时数组，重新赋值即可

注意细节：vector要指定大小后才能[]访问

```c++
#include<iostream>
#include<vector>
using namespace std;

void shuffle(int n,int k,vector<int> &v)
{
    while(k--)
    {
        vector<int> temp=v;
        for(int i=0;i<n;++i)
        {
            v[2*i] = temp[i];
            v[2*i+1] = temp[i+n];
        }
    }
}

int main()
{
    int t;
    int n;
    int k;
    while(cin >> t)
    {
        while(t--)
        {
            cin >> n >> k;
            vector<int> v(n*2);
            for(int i = 0; i < 2*n;++i)
            {
                cin >> v[i];
            }
            shuffle(n,k,v);
            for(auto& e:v)
            {
                cout << e<<" ";
            }
            cout << endl;
        }
    }
    return 0;
}
```
